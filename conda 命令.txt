当前conda环境配置：
python = 3.8.19
pytorch = 2.2.2
pytorch-cuda = 12.1


conda env list :查看所有虚拟环境
conda activate env_name :激活虚拟环境
conda deactivate :退出当前环境
conda list :展示当前环境下所有的包
conda remove --name env_name --all :删除虚拟环境
conda install package_name :安装包
conda uninstall package_name :卸载包
conda create -n new_env_name --clone old_env_name :复制环境
conda create -n GNN-code --clone GNN-master :复制环境
conda env export > environment.yml :打包环境 (仅打包名称，新设备仍需联网安装包工具)
conda env create -f environment.yml ：在新机器上创建环境
pip install conda-pack : 安装离线打包
conda pack -n env_name : 在旧设备上打包压缩包
tar -xzvf env_name.tar.gz -C path\env_name : 新环境下解压到目的文件目录下


git config --global --get http.proxy
git config --global --get https.proxy ：查看代理

git config --global --unset http.proxy
git config --global --unset https.proxy ：git取消代理
